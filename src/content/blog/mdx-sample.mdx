TypeScript:
```ts
interface Post {
  title: string;
  pubDate: Date;
  tags: string[];
}

const getLatest = (posts: Post[]): Post => {
  return posts.sort((a, b) => b.pubDate.getTime() - a.pubDate.getTime())[0];
};
```
